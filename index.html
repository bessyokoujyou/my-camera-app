const SPREADSHEET_ID = "1RFX22MX4iWc2Vwh2ykAnApvQ3TTupUFikr_gTgh3qxQ"; // Webアプリへのアクセス（GET通信）を処理 function doGet(e) { const action = e.parameter.action; if (action === "ping") { return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "pong" })).setMimeType(ContentService.MimeType.JSON); } if (action === "getShiji") { return getShijiData(); } if (action === "getOne") { return getOneOrder(e.parameter.orderNo); } return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Invalid action" })).setMimeType(ContentService.MimeType.JSON); } // データ送信（POST通信）を処理 function doPost(e) { try { const postData = JSON.parse(e.postData.contents); if (postData.action === "save") { return saveNippo(postData.data); } } catch(error) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // 【スマホ連携】新・日報保存システム（同じ発注Noなら1行に開始と終了をまとめる） function saveNippo(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheets().find(s => s.getSheetId() === 1793578510); if (!sheet) { sheet = ss.getSheetByName("日報集計") || ss.insertSheet("日報集計"); // 初めてシートを作る場合はヘッダーをセット sheet.appendRow(["記録日時", "発注No", "生産ライン場所", "作業人数", "開始時刻", "終了時刻", "稼働時間(分)"]); } const orderNo = String(data.orderNo || "").trim(); const line = data.line || ""; const workers = data.workers || 1; const type = data.type || ""; // 「開始」または「終了」 const now = new Date(); const values = sheet.getDataRange().getValues(); if (type === "開始") { // 「開始」の場合は、新しい行として日報に追記する const rowData = [ now, // A: 記録日時 orderNo, // B: 発注No line, // C: 生産ライン場所 workers, // D: 作業人数 now, // E: 開始時刻 "", // F: 終了時刻（最初は空っぽ） "" // G: 稼稼働時間（最初は空っぽ） ]; sheet.appendRow(rowData); return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "Start recorded" })).setMimeType(ContentService.MimeType.JSON); } else if (type === "終了") { // 「終了」の場合は、同じ発注Noで「終了時刻がまだ書かれていない行」を下から探して書き込む for (let i = values.length - 1; i >= 1; i--) { if (String(values[i][1]).trim() === orderNo && values[i][5] === "") { // 終了時刻をF列（6番目）に書き込み sheet.getRange(i + 1, 6).setValue(now); // 【自動連動計算】開始時刻から何分かかったかを計算してG列（7番目）に書き込み const startTime = new Date(values[i][4]); const diffMinutes = Math.round((now - startTime) / 1000 / 60); sheet.getRange(i + 1, 7).setValue(diffMinutes); return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "End recorded and calculated" })).setMimeType(ContentService.MimeType.JSON); } } // 万が一、開始データが見つからなかった場合は新規行として終了だけ記録 sheet.appendRow([now, orderNo, line, workers, "", now, ""]); return ContentService.createTextOutput(JSON.stringify({ status: "success", message: "End recorded as new row" })).setMimeType(ContentService.MimeType.JSON); } } // 加工指図残からデータ取得 (gid: 85631405) function getShijiData() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheets().find(s => s.getSheetId() === 85631405) || ss.getSheets()[0]; }); return obj; }); return ContentService.createTextOutput(JSON.stringify({ status: "success", data: result })).setMimeType(ContentService.MimeType.JSON); } // 発注Noから1件取得 function getOneOrder(orderNo) { if (!orderNo) return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "Required" })).setMimeType(ContentService.MimeType.JSON);
